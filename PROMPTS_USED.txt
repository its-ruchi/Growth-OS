import { GoogleGenAI, ThinkingLevel } from '@google/genai';
import { SHARED_SYSTEM_PROMPT, UNIVERSAL_USER_TEMPLATE, AGENT_PROMPTS } from '../src/constants';
import type { AgentId, AgentResponse } from '../src/types';
import { config } from './config';

class GeminiService {
  private ai: GoogleGenAI;

  constructor(apiKey: string) {
    this.ai = new GoogleGenAI({ apiKey });
  }

  private pickModel(agentId: AgentId) {
    // Use Flash by default for speed + broad availability.
    // Pro can be enabled explicitly if the project has quota for it.
    const useProForDrafts = process.env.GEMINI_USE_PRO_FOR_DRAFTS === 'true';

    if (agentId === 'post_drafter' && useProForDrafts) return 'gemini-3.1-pro-preview';
    return 'gemini-3-flash-preview';
  }

  async runAgent<T>(agentId: AgentId, payload: unknown): Promise<AgentResponse<T>> {
    const agentPrompt = AGENT_PROMPTS[agentId];
    if (!agentPrompt) {
      throw new Error(`No prompt found for agent: ${agentId}`);
    }

    const systemInstruction = `${SHARED_SYSTEM_PROMPT}\n${agentPrompt}`;
    const userPrompt = UNIVERSAL_USER_TEMPLATE
      .replace('{{agent_id}}', agentId)
      .replace('{{json_payload}}', JSON.stringify(payload, null, 2));

    const model = this.pickModel(agentId);

    try {
      const generate = async (selectedModel: string) => {
        return await this.ai.models.generateContent({
          model: selectedModel,
          contents: userPrompt,
          config: {
            systemInstruction,
            responseMimeType: 'application/json',
            // Keep latency predictable.
            thinkingConfig: {
              thinkingLevel: ThinkingLevel.LOW,
            },
          },
        });
      };

      let response;
      try {
        response = await generate(model);
      } catch (e) {
        // If Pro quota/rate limit is exhausted, automatically fall back to Flash.
        const msg = e instanceof Error ? e.message : String(e);
        if (model.includes('pro') && /RESOURCE_EXHAUSTED|quota|rate/i.test(msg)) {
          response = await generate('gemini-3-flash-preview');
        } else {
          throw e;
        }
      }

      const text = response.text;
      if (!text) {
        throw new Error('Empty response from Gemini');
      }

      let jsonStr = text.trim();
      if (jsonStr.startsWith('```')) {
        const match = jsonStr.match(/```(?:json)?\s*([\s\S]*?)\s*```/);
        if (match) jsonStr = match[1].trim();
      }

      const firstChar = jsonStr[0];
      if (firstChar === '{' || firstChar === '[') {
        const closingChar = firstChar === '{' ? '}' : ']';
        let openCount = 0;
        let lastMatchIndex = -1;

        for (let i = 0; i < jsonStr.length; i++) {
          if (jsonStr[i] === firstChar) openCount++;
          else if (jsonStr[i] === closingChar) {
            openCount--;
            if (openCount === 0) {
              lastMatchIndex = i;
              break;
            }
          }
        }

        if (lastMatchIndex !== -1) {
          jsonStr = jsonStr.slice(0, lastMatchIndex + 1);
        }
      }

      return JSON.parse(jsonStr) as AgentResponse<T>;
    } catch (error) {
      return {
        ok: false,
        agent: agentId,
        version: 'v0.1',
        telemetry: [],
        data: {} as T,
        warnings: [],
        next_actions: [],
        error: {
          code: 'AGENT_ERROR',
          message:
            error instanceof Error ? error.message : 'Unknown error occurred',
        },
      };
    }
  }
}

export const geminiService = new GeminiService(config.geminiApiKey);
